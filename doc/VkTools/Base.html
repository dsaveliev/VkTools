def log_exception(exc)
  return unless !!VkTools.logger
  exc = Exception.new(exc) if String === exc
  VkTools.logger.error "VkTools. " + exc.class.to_s + " : " + exc.message + "\n" + exc.backtrace.join("\n")
end
