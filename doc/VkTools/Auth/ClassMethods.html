def authorize(login, password)
  auth_data = inner_authorize(login, password)
  vk_api = VkTools::Api.new :access_token => auth_data[:access_token]
  vk_pages = VkTools::Pages.new :cookie => auth_data[:cookie]
  if block_given?
    yield vk_api, vk_pages 
  else
    [vk_api, vk_pages]
  end
end
